class CheckPrecision:
    def check_precision(self, value):
        if isinstance(value, Decimal):
            if len(value.as_string().split('.')[-1]) > 1:
                result = value.quantize(Decimal("0."), context=Decimal.ROUND_DOWN)
                return result
        else:
            return value

data = {
