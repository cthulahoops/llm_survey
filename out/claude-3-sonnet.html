from decimal import Decimal
from typing import List

from marshmallow import fields, post_load
from marshmallow_dataclass import dataclass as marshmallow_dataclass

@dataclass

    amount: Decimal

    @classmethod
    @marshmallow_dataclass(Order)
    class OrderSchema(marshmallow.Schema):
        created = fields.DateTime()
        amount = fields.Decimal()

@dataclass
class Customer:
    first_name: str
    last_name: str
    date_of_birth: date
    orders: List[Order]

    @classmethod
    @marshmallow_dataclass(Customer)
    class CustomerSchema(marshmallow.Schema):
        first_name = fields.Str()
        last_name = fields.Str()
        date_of_birth = fields.Date()
        orders = fields.Nested(Order.OrderSchema, many=True)

json_data = {
    'first_name': 'Jane',
    'last_name': 'Smith',

    'orders': [
        {
            'created': '2024-01-25T15:00:04',
            'amount': '100.00'
        }
    ]
}

schema = Customer.CustomerSchema()
customer = schema.load(json_data)

print(customer)

In this example, we define two dataclasses: Order and Customer. Each dataclass has a corresponding schema that defines the structure of the data.

  • For the Customer dataclass, we define first_name and last_name as str fields, date_of_birth as a date field, and orders as a nested list of Order objects.

The marshmallow_dataclass decorator is used to create the schemas from the dataclasses automatically.
