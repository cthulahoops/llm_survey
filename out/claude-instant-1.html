Prompt

I need to take this json:

{
'first_name': 'Jane',
'last_name': 'Smith',
'date_of_birth': '03/16/1977',
orders: [
{
'created': '2024-01-25T15:00:04',
'amount': '100.00'
}
]}

And generate some dataclasses.

date_of_birth needs to be a date.
created is a datetime.
amount is a decimal.

How would I do this using the marshmallow library in python?

Response 1.

Here is how you can define dataclass models and marshmallow schemas to serialize/deserialize that JSON into Python objects with the correct data types:

from dataclasses import dataclass
from decimal import Decimal
from datetime import date, datetime

from marshmallow import Schema, fields

# Dataclass models
@dataclass
class Order:
    created: datetime
    amount: Decimal

@dataclass
class Person:
    first_name: str 
    last_name: str
    date_of_birth: date
    orders: List[Order]

# Marshmallow schemas    
class OrderSchema(Schema):
    created = fields.DateTime()
    amount = fields.Decimal()

class PersonSchema(Schema):
    first_name = fields.Str()
    last_name = fields.Str()  
    date_of_birth = fields.Date()
    orders = fields.Nested(OrderSchema, many=True)

# Deserialize JSON to object
schema = PersonSchema()
person = schema.load(json_data)

# person will now have the correct types
print(type(person.date_of_birth)) # <class 'datetime.date'>

The key points:

  • Define dataclass models with correct types
  • Define marshmallow schemas that match models
  • Use fields.DateTime(), fields.Decimal() etc to specify types
  • Call schema.load() to deserialize JSON into object

This will deserialize the JSON into Python objects with the data serialized to the correct data types.

from dataclasses import dataclass
from datetime import date, datetime
from decimal import Decimal

from marshmallow import Schema, fields, pre_load

@dataclass
class Order:
    created: datetime
    amount: Decimal

@dataclass
class Person:
    first_name: str
    last_name: str
    date_of_birth: date
    orders: list[Order]

class OrderSchema(Schema):
    created = fields.DateTime()
    amount = fields.Decimal()

class PersonSchema(Schema):
    first_name = fields.Str()
    last_name = fields.Str()

    @pre_load
    def parse_date(self, data):
        data['date_of_birth'] = datetime.strptime(data['date_of_birth'], '%m/%d/%Y').date()
        return data

    date_of_birth = fields.Date()
    orders = fields.Nested(OrderSchema, many=True)

# Deserialize
schema = PersonSchema()
data = schema.loads(json_string)

# person will be a Person dataclass instance
person = data.data

Here is how you can generate dataclasses from that JSON using Marshmallow:

from dataclasses import dataclass
from datetime import datetime
from decimal import Decimal
from marshmallow import Schema, fields, post_load

@dataclass
class Order:
    created: datetime
    amount: Decimal

@dataclass
class Person:
    first_name: str 
    last_name: str
    date_of_birth: date
    orders: list

class OrderSchema(Schema):
    created = fields.DateTime()
    amount = fields.Decimal()

    @post_load
    def make_object(self, data):
        return Order(**data)

class PersonSchema(Schema):
    first_name = fields.Str()
    last_name = fields.Str()
    date_of_birth = fields.Date()
    orders = fields.Nested(OrderSchema, many=True)

    @post_load
    def make_object(self, data):
        return Person(**data)

schema = PersonSchema()
result = schema.load({
  'first_name': 'Jane',
  'last_name': 'Smith', 
  'date_of_birth': '03/16/1977',
  'orders': [
    {'created': '2024-01-25T15:00:04', 'amount': '100.00'}
  ]
})

print(result)

This will deserialize the JSON into dataclass instances with the correct types for date/datetime/decimal fields.
