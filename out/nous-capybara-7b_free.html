import datetime
import marshmallow as mm

class Order(mm.Schema):
    created = mm.datetime(format='iso8601')
    amount = mm.decimal(precision=2)

    class Meta:
        unknown = mm.ignore()

class Customer(mm.Schema):
    first_name = mm.str()
    last_name = mm.str()
    date_of_birth = mm.datetime(format='iso8601')
    orders = mm.List(order=Order())

    class Meta:
        unknown = mm.ignore()

from customer import Customer

      "created": "2024-01-25T15:00:04",
      "amount": "100.00"
    }
  ]
}
'''

customer_data = Customer(**Customer.Meta.unknown)
customer_data.load(json_string)
