Here's what the code would look like:

from decimal import Decimal
from dataclasses import dataclass
from datetime import datetime, date
from typing import List
from marshmallow import Schema, fields, post_load

@dataclass
class Order:
    created: datetime
    amount: Decimal

@dataclass
class Person:
    first_name: str
    last_name: str
    date_of_birth: date
    orders: List[Order]

class OrderSchema(Schema):
    created = fields.DateTime(required=True)
    amount = fields.Decimal(as_string=True)

    @post_load
    def make_order(self, data, **kwargs):
        return Order(**data)

class PersonSchema(Schema):
    first_name = fields.Str()
    last_name = fields.Str()
    date_of_birth = fields.Date()
    orders = fields.List(fields.Nested(OrderSchema))

    @post_load
    def make_person(self, data, **kwargs):
        return Person(**data)

# Now you can use the schemas to load/dump data
person_schema = PersonSchema()

# Let's assume we have some JSON data
json_data = {
    'first_name': 'Jane',
    'last_name': 'Smith',
    'date_of_birth': '03/16/1977',
    'orders': [
        {
            'created': '2024-01-25T15:00:04',
            'amount': '100.00'
        }
    ]
}
# Deserialize the JSON data to Python dataclass
person = person_schema.load(json_data)
print(person)

The make_order and make_person methods decorated with @post_load are a part of marshmallow's deserialization process, which ensures the data is loaded into the dataclass after being deserialized.

Note: as_string=True is passed to fields.Decimal because if you were to load the value straight from JSON, you would get a float, which could cause precision issues. This way, the value is loaded as a string and converted to Decimal, keeping precision intact.
