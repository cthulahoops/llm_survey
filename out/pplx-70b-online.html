I need to take this json:

'first_name': 'Jane',
'last_name': 'Smith',
'date_of_birth': '03/16/1977',

{
'created': '2024-01-25T15:00:04',
'amount': '100.00'
}
]}

And generate some dataclasses.

date_of_birth needs to be a date.
created is a datetime.
amount is a decimal.

How would I do this using the marshmallow library in python?

To generate dataclasses and corresponding Marshmallow schemas using marshmallow_dataclass for the JSON provided:

import datetime
from marshmallow_dataclass import dataclass, field_for_schema

# Modify your JSON slightly to be a data structure
data = {
  'first_name': 'Jane',
  'last_name': 'Smith',
  'date_of_birth': '03/16/1977',
  'orders': [
    {
      'created': '2024-01-25T15:00:04',
      'amount': '100.00'
    }
  ]
}

# Define the dataclasses
@dataclass
class Order:
    created: datetime.datetime
    amount: float

@dataclass
class Person:
    first_name: str
    last_name: str
    date_of_birth: datetime.date
    orders: List[Order]

# Generate Marshmallow schemas
from marshmallow import fields
from marshmallow_dataclass import field_from_dataclass

OrderSchema = field_from_dataclass(Order, field_fallback=fields.FIELD_LOADER)
PersonSchema = field_from_dataclass(Person, field_fallback=fields.FIELD_LOADER)

# The schema for Orders as a List is automatically handled by Marshmallow
# There's no need to specify the type in PersonSchema since it is generated for you.

# Example usage
example_person = Person(
    first_name="Jane",
    last_name="Smith",
    date_of_birth=datetime.date(1977, 3, 16),
    orders=[
        Order(
            created=datetime.datetime(2024, 1, 25, 15, 0, 0),
            amount=100.0
        )
    ],
)

serialized_person = PersonSchema().dump(example_person)
print(serialized_person)

# Outputs: {"first_name": "Jane", "last_name": "Smith", "date_of_birth": "1977-03-16", "orders": [{"created": "2024-01-25T15:00:00", "amount": 100.0}]}

from dataclasses import dataclass
from datetime import datetime, date
from decimal import Decimal
from marshmallow import Schema, fields, post_load

@dataclass
class Person:
first_name: str
last_name: str
date_of_birth: date
orders: list

class PersonSchema(Schema):
first_name = fields.Str()
last_name = fields.Str()
date_of_birth = fields.Date()
orders = fields.Nested(OrderSchema, many=True)

class OrderSchema(Schema):
created = fields.DateTime()
amount = fields.Decimal(default_precision=2)

@post_load
def make_person(self, data, kwargs):
return Person(data)

# Code to handle your JSON
json_data = {
"first_name": "Jane",
"last_name": "Smith",

"amount": "100.00"
}
]
}

person_schema = PersonSchema()
person_data = person_schema.load(json_data)
person = person_schema.make_person(person_data)
print(person) # The Person dataclass instance

PersonSchema and OrderSchema are Marshmallow schemas that help convert your JSON to an actual dataclass instance. The post_load function is a Marshmallow feature that allows you to transform the result dict into an actual dataclass instance.
