from datetime import date, datetime
from decimal import Decimal
from marshmallow import Schema, fields

class OrderSchema(Schema):
    created = fields.DateTime()
    amount = fields.Decimal(as_string=True)

class UserSchema(Schema):
    first_name = fields.String()
    last_name = fields.String()
    date_of_birth = fields.Date()
    orders = fields.Nested(OrderSchema, many=True)

# JSON data
data = {
    'first_name': 'Jane',
    'last_name': 'Smith',
    'date_of_birth': '03/16/1977',
    'orders': [
        {
            'created': '2024-01-25T15:00:04',
            'amount': '100.00'
        }
    ]
}

# Load JSON data using the schema
user_schema = UserSchema()
result = user_schema.load(data)

# Access the data as dataclasses
user_data = result.data
orders_data = result.data.orders

print(user_data)
print(orders_data)

In the above code, we create two schemas: OrderSchema and UserSchema. OrderSchema defines the schema for the orders field, and UserSchema defines the schema for the overall JSON. The fields in OrderSchema are defined as DateTime() and Decimal(as_string=True) to match the created and amount types specified.

We then load the JSON data into dataclasses using the load method of UserSchema. The resulting data can be accessed as dataclasses.
