import marshmallow as ma
from datetime import date, datetime

class DateSchema(ma.Schema):
    class Meta:
        type_ = date

class DecimalSchema(ma.Schema):
    class Meta:
        type_ = float

class OrderSchema(ma.Schema):
    created = ma.fields.DateTime(as_datetime=True)
    amount = ma.fields.Decimal()

class UserSchema(ma.Schema):
    first_name = ma.fields.Str()
    last_name = ma.fields.Str()
    date_of_birth = DateSchema()
    orders = ma.fields.Nested(OrderSchema, many=True)

class User:
    def __init__(self, first_name, last_name, date_of_birth, orders):
        self.first_name = first_name
        self.last_name = last_name
        self.date_of_birth = date_of_birth
        self.orders = orders

    class Config:
        schema_class = UserSchema

if __name__ == '__main__':
    user_dict = {
        'first_name': 'Jane',

Response 3.

Here's a basic example of how you could use the marshmallow library to deserialize the provided JSON into dataclasses:

from datetime import datetime
from dataclasses import dataclass
from marshmallow import Schema, fields, post_load
import dateutil.parser

@dataclass
class Order:
    created: datetime
    amount: decimal

class OrderSchema(Schema):
    created = fields.Field(load=dateutil.parser.parse)
    amount = fields.Decimal()

    @post_load
    def make_order(self, data):
        return Order(**data)

@dataclass
class User:
    first_name: str
    last_name: str
    date_of_birth: datetime
    orders: list[Order]

class UserSchema(Schema):
    first_name = fields.Str()
    last_name = fields.Str()
    date_of_birth = fields.Field(load=dateutil.parser.parse)
    orders = fields.Nested(OrderSchema, many=True)

    @post_load
    def make_user(self, data):
        return User(**data)

user_dict = {
  'first_name': 'Jane',
  'last_name': 'Smith',
  'date_of_birth': '03/16/1977',
  'orders': [
    {
      'created': '2024-01-25T15:00:04',
      'amount': '100.00'
    }
]}

user_schema = UserSchema()
user = user_schema.load(user_dict)

print(user.first_name) # Jane
print(user.last_name) # Smith
print(user.date_of_birth) # <datetime obj>
print(user.orders[0].created) # <datetime obj>
print(user.orders[0].amount) # 100.0

In this example, I've defined the User and Order dataclasses, as well as their respective schemas (using the Marshmallow library). Then, I've used the load() method of the UserSchema class to deserialize the provided JSON into a User object.

The post_load decorator is used to convert the raw data from the JSON into the actual dataclass objects.

Note that you'll need to import dateutil for the date parsing to work.
