from dataclasses import dataclass
from marshmallow import Schema, fields, ValidationError, validates, pre_dump, pre_load
from marshmallow.class_ import class_metadata
from marshmallow.class_ import class_schema
from datetime import date, datetime
from decimal import Decimal

# Define the dataclasses
@dataclass
class Order:
    created: datetime
    amount: Decimal

@dataclass
class Person:
    first_name: str
    last_name: str
    date_of_birth: date
    orders: list[Order]

# Define the schemas
class OrderSchema(Schema):
    created = fields.DateTime(required=True)
    amount = fields.Decimal(required=True)

class PersonSchema(Schema):
    first_name = fields.Str(required=True)
    last_name = fields.Str(required=True)
    date_of_birth = fields.Date(required=True)
    orders = fields.Nested(OrderSchema, required=True)

    @validates('date_of_birth')

        if value > date.today():
            raise ValidationError('Date of birth cannot be in the future')

    @pre_dump(pass_many=False)

    @pre_load
    def process_orders(self, data, **kwargs):
        data['orders'] = self.process_class_data(Order, data.get('orders'))

# Get the schema for the Person dataclass
PersonDataclassSchema = class_schema(Person, schema_cls=PersonSchema)
