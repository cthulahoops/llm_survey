Prompt

I need to take this json:

{
'first_name': 'Jane',
'last_name': 'Smith',
'date_of_birth': '03/16/1977',
orders: [
{
'created': '2024-01-25T15:00:04',
'amount': '100.00'
}
]}

And generate some dataclasses.

date_of_birth needs to be a date.
created is a datetime.
amount is a decimal.

How would I do this using the marshmallow library in python?

```python
from marshmallow import Schema, fields

class OrderSchema(Schema):
created = fields.DateTime()
amount = fields.Decimal(as_string=True)

class UserSchema(Schema):
first_name = fields.String()
last_name = fields.String()
date_of_birth = fields.Date()
orders = fields.Nested(OrderSchema, many=True)

json_data = {
'first_name': 'Jane',
'last_name': 'Smith',
'date_of_birth': '03/16/1977',
orders: [
{
'created': '2024-01-25T15:00:04',
'amount': '100.00'
}
]
}

schema = UserSchema()
user = schema.load(json_data).data

for order in user.orders:
print(order.created)
print(order.amount)
```

```python
from dataclasses import dataclass
from datetime import date, datetime
from decimal import Decimal
import marshmallow_dataclass
import marshmallow
from marshmallow import fields
import json

json_data = json.loads("""
{
'first_name': 'Jane',
'last_name': 'Smith',
'date_of_birth': '03/16/1977',
orders: [
{
'created': '2024-01-25T15:00:04',
'amount': '100.00'
}
]}
""")

@dataclass
class Order:
created: datetime
amount: Decimal

@dataclass
class User:
first_name: str
last_name: str
date_of_birth: date
orders: list[Order]

class DateField(fields.Field):
def _deserialize(self, value, **kwargs):
return datetime.strptime(value, '%m/%d/%Y').date()

class DateTimeField(fields.Field):
def _deserialize(self, value, **kwargs):
return datetime.strptime(value, '%Y-%m-%dT%H:%M:%S')

class DecimalField(fields.Field):
def _deserialize(self, value, **kwargs):
return Decimal(value)

UserSchema = marshmallow_dataclass.class_schema(User)
UserSchema.Meta.datefields = ['date_of_birth']
UserSchema.Meta.fields['date_of_birth'] = DateField()
UserSchema.Meta.fields['created'] = DateTimeField()
UserSchema.Meta.fields['amount'] = DecimalField()

user_schema = UserSchema()
user = user_schema.load(json_data)
print(user)
```
